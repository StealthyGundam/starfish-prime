function enableInputs() { const lines = terminal.querySelectorAll(".terminal-line"); let userKeywordAttached = false; lines.forEach(line => { /* ADMIN — aesthetic only */ if (line.textContent.includes("| >> Encryption Key:") && line.textContent.includes("Administrator")) { line.textContent = "| >> Encryption Key: ■■■"; } if (line.textContent === "| >> Keyword:" && !userKeywordAttached) { line.textContent = "| >> Keyword: ■■■■■■■■■■"; } /* USER — functional */ if (line.textContent.includes("| >> Decryption Key:")) { line.textContent = "| >> Decryption Key: "; const s = document.createElement("span"); s.id = "user-key"; s.className = "input-placeholder"; s.contentEditable = true; s.textContent = "45"; s.addEventListener("input", update); line.appendChild(s); } if (line.textContent === "| >> Keyword:" && !document.getElementById("user-keyword")) { userKeywordAttached = true; line.textContent = "| >> Keyword: "; const s = document.createElement("span"); s.id = "user-keyword"; s.className = "input-placeholder"; s.contentEditable = true; s.textContent = "BAMBOOZLE SWIPE"; s.addEventListener("input", update); line.appendChild(s); } }); /* Force initial render */ update(); }
